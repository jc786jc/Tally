const TAG_COLORS = {
  NEW:     { bg: "#fff3cd", text: "#856404", border: "#f0ad4e" },
  UPDATED: { bg: "#d1ecf1", text: "#0c5460", border: "#bee5eb" },
};

const TableCard = ({ table }) => {
  const [expanded, setExpanded] = useState(true);
  const tag = table.tag ? TAG_COLORS[table.tag] : null;

  const colHeaders = ["Column Name", "Data Type", "Description", "Required", "Source", "Source Column Logic", "Key"];

  return (
    <div style={{
      border: "1px solid #dee2e6", borderRadius: 8,
      overflow: "hidden", marginBottom: 16,
      boxShadow: "0 1px 6px rgba(0,0,0,0.07)",
    }}>
      {/* Header */}
      <div
        onClick={() => setExpanded(!expanded)}
        style={{
          background: tag ? tag.bg : "#f8f9fa",
          padding: "12px 16px", cursor: "pointer",
          display: "flex", alignItems: "center", gap: 10,
          borderBottom: expanded ? "1px solid #dee2e6" : "none",
        }}>
        <div style={{ flex: 1 }}>
          <div style={{ display: "flex", alignItems: "center", gap: 8, marginBottom: 4 }}>
            <span style={{ fontSize: 13, fontWeight: 700, color: "#d9534f", fontFamily: "monospace" }}>
              {table.name}
            </span>
            {table.tag && (
              <span style={{
                fontSize: 9, fontWeight: 700, padding: "2px 7px", borderRadius: 4,
                background: tag.bg, color: tag.text, border: `1px solid ${tag.border}`,
              }}>{table.tag}</span>
            )}
            <span style={{
              fontSize: 10, color: "#6c757d", padding: "2px 8px",
              background: "white", borderRadius: 4, border: "1px solid #dee2e6",
            }}>{table.module}</span>
            <span style={{ fontSize: 10, color: "#6c757d" }}>
              {table.cols.length} columns
            </span>
          </div>
          <div style={{ fontSize: 11, color: "#6c757d", fontStyle: "italic" }}>{table.desc}</div>
        </div>
        <span style={{ fontSize: 12, color: "#aaa" }}>{expanded ? "▲" : "▼"}</span>
      </div>

      {/* Columns */}
      {expanded && (
        <div style={{ overflowX: "auto" }}>
          <table style={{ width: "100%", borderCollapse: "collapse", fontSize: 11 }}>
            <thead>
              <tr style={{ background: "#2c3e50" }}>
                {colHeaders.map((h, i) => (
                  <th key={i} style={{
                    padding: "8px 12px", textAlign: "left",
                    fontWeight: 600, color: "white", fontSize: 10,
                    letterSpacing: 0.5, whiteSpace: "nowrap",
                    borderRight: "1px solid #3d5166",
                  }}>{h}</th>
                ))}
              </tr>
            </thead>
            <tbody>
              {table.cols.map((col, i) => (
                <tr key={i} style={{
                  background: i % 2 === 0 ? "white" : "#f8f9fa",
                  borderTop: "1px solid #f1f3f5",
                }}>
                  {/* Column Name */}
                  <td style={{
                    padding: "7px 12px",
                    fontFamily: "monospace",
                    fontWeight: 700,
                    color: col.key === "PK" ? "#d9534f" :
                           col.key.includes("FK") ? "#27ae60" : "#333",
                    whiteSpace: "nowrap",
                    borderRight: "1px solid #f1f3f5",
                  }}>{col.name}</td>

                  {/* Data Type */}
                  <td style={{
                    padding: "7px 12px",
                    fontFamily: "monospace",
                    color: "#6c757d",
                    whiteSpace: "nowrap",
                    borderRight: "1px solid #f1f3f5",
                  }}>{col.type}</td>

                  {/* Description */}
                  <td style={{
                    padding: "7px 12px",
                    color: "#333",
                    borderRight: "1px solid #f1f3f5",
                    minWidth: 200,
                  }}>{col.desc}</td>

                  {/* Required */}
                  <td style={{
                    padding: "7px 12px",
                    textAlign: "center",
                    whiteSpace: "nowrap",
                    borderRight: "1px solid #f1f3f5",
                  }}>
                    <span style={{
                      fontSize: 9, fontWeight: 700,
                      padding: "2px 7px", borderRadius: 4,
                      background: col.required === "Yes" ? "#f8d7da" : "#e2e3e5",
                      color: col.required === "Yes" ? "#721c24" : "#383d41",
                    }}>{col.required}</span>
                  </td>

                  {/* Source */}
                  <td style={{
                    padding: "7px 12px",
                    fontFamily: "monospace",
                    fontSize: 10,
                    color: col.source === "System" ? "#2980b9" :
                           col.source === "Frontend" ? "#8e44ad" : "#27ae60",
                    whiteSpace: "nowrap",
                    fontWeight: 600,
                    borderRight: "1px solid #f1f3f5",
                  }}>{col.source}</td>

                  {/* Source Column Logic */}
                  <td style={{
                    padding: "7px 12px",
                    color: "#555",
                    fontStyle: "italic",
                    fontSize: 10,
                    borderRight: "1px solid #f1f3f5",
                    minWidth: 280,
                  }}>{col.logic}</td>

                  {/* Key */}
                  <td style={{
                    padding: "7px 12px",
                    textAlign: "center",
                    whiteSpace: "nowrap",
                  }}>
                    {col.key && (
                      <span style={{
                        fontSize: 9, fontWeight: 700,
                        padding: "2px 7px", borderRadius: 4,
                        background: col.key === "PK" ? "#f8d7da" :
                                    col.key.includes("FK") ? "#d4edda" : "#e2e3e5",
                        color: col.key === "PK" ? "#721c24" :
                               col.key.includes("FK") ? "#155724" : "#383d41",
                      }}>{col.key}</span>
                    )}
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      )}
    </div>
  );
};
